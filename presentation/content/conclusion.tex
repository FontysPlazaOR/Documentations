\subsection{Pro's and Con's}
\frame{
	\frametitle{Pro's}
	
	\begin{itemize}
		\item Vuzix use Android as OS and nothing new.
		\item Software is running on Microsoft Windows 10
		\item Uses Eclipse as IDE
	\end{itemize}	
}

\frame{
	\frametitle{Con's}
	
	\begin{itemize}
		\item Flashing and working with Vuzix needs Microsoft Windows
		\item Vuzix webpage is mostly unreachable and shows only 404-Pages
		\item Only less informations in the internet
	\end{itemize}	
}
